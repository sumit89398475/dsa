function sortArrayAscending(arr) { let len = arr.length; for (let i = 0; i < len; i++) { for (let j = 0; j < len - 1 - i; j++) { if (arr[j] > arr[j + 1]) { // Swap elements let temp = arr[j]; arr[j] = arr[j + 1]; arr[j + 1] = temp; } } } return arr; } // Example usage: const numbers = [5, 2, 9, 1, 5, 6]; const sortedNumbers = sortArrayAscending(numbers); console.log(sortedNumbers); // Output: [1, 2, 5, 5, 6, 9]
